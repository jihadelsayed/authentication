@Component({
  selector: 'app-sign-up',
  templateUrl: './sign-up.component.html',
  styleUrls: ['./sign-up.component.scss'],
  standalone: true,

      this.resetForm();

        this.host = params['host'] ?? null;
        this.port = params['port'] ?? '443';
        this.pathname = params['pathname'] ?? '';
        this.language = params['language'] ?? 'en';
      });
    
    this.resetForm();

   {

    
     if(form!=null)
